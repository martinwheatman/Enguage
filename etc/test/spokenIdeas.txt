# First, check we're clean!
hello:                          I don't understand, hello.

# Build a repertoire...
to the phrase hello reply hello to you too: ok.
hello:                          hello to you too.
to the phrase my name is variable name reply hello variable name: ok.

# ... and save this...
save  spoken concepts as hello: ok.

hello: hello to you too.
my name is martin: hello martin.
			
# then remove the cached concept...
unload hello.
			
# So, this will reload from the saved repertoire
hello:                          hello to you too.
			
# OK, now delete the repertoire (and the sign)
delete spoken concept hello:    ok.

# ... its gone, and we're clean again!
hello:                          i don't understand, hello.
