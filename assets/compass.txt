#
# Compass - managing the on/off state of a Unity3d asset
#

# We want to turn the compass on, or turn it off...:
On "compass ONOROFF":
	#http localhost:6578 "<asset id='compass' op='ONOROFF'/>";
	perform "variable set compass ONOROFF";
	then, reply "ok, compass is ONOROFF".	
On "turn ONOROFF     compass", compass ONOROFF.
On "turn ONOROFF the compass", compass ONOROFF.
On "turn     compass ONOROFF", compass ONOROFF.
On "turn the compass ONOROFF", compass ONOROFF.

On "the compass is ONOROFF":
	perform "variable exists compass ONOROFF";
	if not, reply "no, the compass is not ONOROFF";
	then, reply "yes, the compass is ONOROFF".
On "is the compass ONOROFF", the compass is ONOROFF.	
On "is     compass ONOROFF", the compass is ONOROFF.	

#..or, toggle its state:
On "toggle compass":
	the compass is on;
	turn compass off;
	if not, turn compass on;
	then, reply "ok".
On "toggle the compass", toggle compass.
On "compass",            toggle compass.
